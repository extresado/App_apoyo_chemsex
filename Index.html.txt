        function renderContent(viewId) {
            const data = drugData[viewId];
            if (!data) return;


            const contentArea = document.getElementById('content-area');


            // Paso 1: Fade out (Opacidad 0) para iniciar la transición
            contentArea.style.opacity = '0';
            contentArea.style.transform = 'translateY(10px)';
            
            // Esperar el fade-out antes de cambiar el contenido (300ms)
            setTimeout(() => {
                currentView = viewId;
                document.getElementById('content-title').innerText = data.title;
                // Renderizar el contenido, pasando el ícono de la sección
                contentArea.innerHTML = data.content(data.icon);
                
                // Paso 2: Actualizar botones de navegación y añadir el "active-button"
                const buttons = document.querySelectorAll('#nav-buttons button');
                buttons.forEach(button => {
                    // Limpiar clases de activo
                    button.classList.remove('active-button');
                    button.classList.add('bg-card', 'text-gray-300', 'hover:bg-gray-700');


                    if (button.dataset.view === viewId) {
                        // Aplicar clases de activo y eliminar las de inactivo
                        button.classList.remove('bg-card', 'text-gray-300', 'hover:
